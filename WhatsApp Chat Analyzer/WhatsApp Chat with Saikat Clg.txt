{
	total=`cat $dbfile | wc -l`
	echo "Total records :"$total

	then
		i=1
		while [ $i -le $total ]
		do
		echo "===================================="
			echo "Employee - "$i
			echo "============="
			record=`cat $dbfile | head -$i | tail -1`
			emp_no=`echo $record | cut -d "|" -f1`
			emp_id=`echo $record | cut -d "|" -f2`

			echo "Employee Designation :" $emp_desg

			i=`expr $i + 1`

		done
